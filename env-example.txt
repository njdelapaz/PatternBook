# Name the file ".env"

# OpenAI API Configuration
# Get your API key from https://platform.openai.com/api-keys
# Used for: AI summaries and transcription (if TRANSCRIBE_PROVIDER=openai)
OPENAI_API_KEY=your_openai_api_key_here

# Transcription Provider Configuration
# Options: 'openai' (default) or 'deepgram'
# If primary provider fails, the app will try the alternate if configured
TRANSCRIBE_PROVIDER=openai

# Deepgram API Configuration (optional - alternative transcription provider)
# Get your API key from https://console.deepgram.com
# Free tier available with generous credits
# Used when TRANSCRIBE_PROVIDER=deepgram or as fallback if OpenAI quota exceeded
DEEPGRAM_API_KEY=your_deepgram_api_key_here
